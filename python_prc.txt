Frequency of chars in string
s="naveennaveen"
result = {}
print(result)
print("loop started")
for char in s:
    print(char, "Letter from ",s)
    if char not in result:
        result[char] = 1
        print(result)
    else:
        result[char] += 1
        print(result)
print("loop ended")
print(result)
